Height of Binary Tree
Medium

Algorithm:
1.most important check in case not a single root exist if(node) return -1
2.first I'll traverse the left side of root node, 
  once we reach end where node have pointing to null i.e if(node==NULL) return ;
3.then I'll traverse the right side of root node
  once we reach end where node have pointing to null i.e if(node==NULL) return ;


and then I'll find max(left,right) + 1


	Height(node* root){
		//check in case not a single root exist
		if(root==NULL) return -1
		
		// return once we reach last node of either left or right side of the tree
		if(root==NULL)
			return ;
		int left = Height(root->left);
		int right = Height(root->right);
		
		int height = max(left, right) + 1;
		
		return height;
	}

    class Solution{
    public:
    //Function to find the height of a binary tree.
    int height(struct Node* node){ 
        // code here 
        if(node==NULL) return 0;
        
        int left = height(node->left);
        int right = height(node->right);
        
        int ans = max(left, right) + 1;
        
        return ans;
    }
};
